üöÄ Getting Started

This RESTful API provides all the functionality needed to build native mobile applications (iOS, Android, React Native, Flutter, etc.) for the Medicine Search System.

Base URL

https://your-domain.com/api/v1

Authentication

The API uses JWT (JSON Web Tokens) for authentication. Include the token in the Authorization header:

Authorization: Bearer YOUR_JWT_TOKEN

üíä Medicine Endpoints

GET /medicines/search?q=paracetamol

Search for medicines by name, composition, or therapeutic use

Query Parameters:

  • q - Search query (required)
  • manufacturer - Filter by manufacturer (optional)
  • discontinued - Filter by status: all, active, discontinued (optional)
GET /medicines/{index}

Get detailed information about a specific medicine

üíª Sample Code

React Native Example

// Search medicines
const searchMedicines = async (query) => {
  try {
    const response = await fetch(
      `https://api.example.com/api/v1/medicines/search?q=${query}`
    );
    const data = await response.json();
    return data.medicines;
  } catch (error) {
    console.error('Error searching medicines:', error);
  }
};

// Login
const login = async (username, password) => {
  try {
    const response = await fetch(
      'https://api.example.com/api/v1/auth/login',
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ username, password }),
      }
    );
    const data = await response.json();
    // Store token in AsyncStorage
    await AsyncStorage.setItem('token', data.token);
    return data;
  } catch (error) {
    console.error('Login error:', error);
  }
};

Flutter/Dart Example

import 'package:http/http.dart' as http;
import 'dart:convert';

// Search medicines
Future> searchMedicines(String query) async {
  final response = await http.get(
    Uri.parse('https://api.example.com/api/v1/medicines/search?q=$query'),
  );
  
  if (response.statusCode == 200) {
    final data = jsonDecode(response.body);
    return (data['medicines'] as List)
        .map((json) => Medicine.fromJson(json))
        .toList();
  } else {
    throw Exception('Failed to load medicines');
  }
}

// Login
Future login(String username, String password) async {
  final response = await http.post(
    Uri.parse('https://api.example.com/api/v1/auth/login'),
    headers: {'Content-Type': 'application/json'},
    body: jsonEncode({
      'username': username,
      'password': password,
    }),
  );
  
  if (response.statusCode == 200) {
    return AuthResponse.fromJson(jsonDecode(response.body));
  } else {
    throw Exception('Login failed');
  }
}

Swift/iOS Example

import Foundation

// Search medicines
func searchMedicines(query: String, completion: @escaping ([Medicine]?) -> Void) {
    let urlString = "https://api.example.com/api/v1/medicines/search?q=\(query)"
    guard let url = URL(string: urlString) else { return }
    
    URLSession.shared.dataTask(with: url) { data, response, error in
        guard let data = data, error == nil else {
            completion(nil)
            return
        }
        
        do {
            let result = try JSONDecoder().decode(SearchResponse.self, from: data)
            completion(result.medicines)
        } catch {
            completion(nil)
        }
    }.resume()
}

// Login
func login(username: String, password: String, completion: @escaping (AuthResponse?) -> Void) {
    let url = URL(string: "https://api.example.com/api/v1/auth/login")!
    var request = URLRequest(url: url)
    request.httpMethod = "POST"
    request.setValue("application/json", forHTTPHeaderField: "Content-Type")
    
    let body: [String: String] = ["username": username, "password": password]
    request.httpBody = try? JSONEncoder().encode(body)
    
    URLSession.shared.dataTask(with: request) { data, response, error in
        guard let data = data, error == nil else {
            completion(nil)
            return
        }
        
        let authResponse = try? JSONDecoder().decode(AuthResponse.self, from: data)
        completion(authResponse)
    }.resume()
}

üìù Best Practices

  • Always store JWT tokens securely (Keychain on iOS, KeyStore on Android)
  • Implement token refresh mechanism for long-lived sessions
  • Handle network errors gracefully with retry logic
  • Cache responses when appropriate to improve performance
  • Use HTTPS for all API requests in production
  • Implement proper error handling for all API calls
  • Add request timeouts to prevent hanging requests
  • Follow platform-specific UI/UX guidelines
